Setup: connect your Google Doc

This page polls a Google Apps Script Web App that reads your Google Doc and returns its plain text. Every keystroke you make in that Doc will appear here within ~2 seconds — for free, with no third-party services.

  1. Open your Google Doc → Extensions → Apps Script
  2. Replace all code with the script below, then click Save
  3. Click Deploy → New deployment → Web app
  4. Set Execute as: Me, Who has access: Anyone
  5. Click Deploy and copy the Web App URL
  6. Paste it below and click Save
// Google Apps Script — paste into your script editor
function doGet() {
  var docId = 'YOUR_GOOGLE_DOC_ID_HERE'; // ← replace this
  var doc  = DocumentApp.openById(docId);
  var body = doc.getBody().getText();
  var output = ContentService
    .createTextOutput(JSON.stringify({ text: body, ts: new Date().toISOString() }))
    .setMimeType(ContentService.MimeType.JSON);
  return output;
}
